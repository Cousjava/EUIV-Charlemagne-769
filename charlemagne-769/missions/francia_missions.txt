vassalize_middle_francia = {
	
	type = country

	category = MIL
	
	allow = {
		normal_or_historical_nations = yes
		tag = Z34
		exists = Z43
		is_subject = no
		dynasty = "Karling"
		french_region = {
			owned_by = ROOT
			is_core = ROOT
		}
		Z43 = {
			is_subject = no
			dynasty = "Karling"
		}
		NOT = { truce_with = Z43 }
		NOT = { war_with = Z43 }
	}
	abort = {
		OR = {
			is_subject = yes
			NOT = { exists = Z43 }
			Z43 = { is_subject = yes }
		}
	}
	success = {
		Z43 = { junior_union_with = Z34 }
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			mil = 4
		}
	}
	immediate = {
		add_casus_belli = {
			type = cb_restore_personal_union
			months = 120
			target = Z43
		}
	}
	abort_effect = {
		remove_casus_belli = {
			type = cb_restore_personal_union
			target = Z43
		}
	}
	effect = {
		add_prestige = 10
		hidden_effect = {
			remove_casus_belli = {
				type = cb_restore_personal_union
				target = Z43
			}
		}
	}
}
